    ux, uy = a.x - b.x, a.y - b.y
    vx, vy = c.x - b.x, c.y - b.y
    du = math.hypot(ux, uy)
    dv = math.hypot(vx, vy)
    if du == 0 or dv == 0:

    rms = math.sqrt((resid**2).sum() / (Xc.shape[0]*Xc.shape[1]))
    return float(rms)
